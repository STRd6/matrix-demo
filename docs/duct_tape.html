Point = require "point"
Size = require "size"
Matrix = require "matrix"

{sqrt} = Math

Matrix.Point = Point

Point::scale = (width, height) ->
  if typeof width is "object"
    {width, height} = width

  height ?= width

  Point(@x * width, @y * height)

Point::abs = ->
  Point(Math.abs(@x), Math.abs(@y))

Matrix::scaleComponent = ->
  Size sqrt(@a * @a + @b * @b), sqrt(@c * @c + @d * @d)

Matrix::translationComponent = ->
  Point @tx, @ty
